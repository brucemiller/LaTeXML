\section{Spacing}
Taller:
\begin{itemize}
\itemsep=2em
\item a
\item b
\item c
\end{itemize}

Taller \& Taller:
\begin{itemize}
\item a
\itemsep=2em
\item b
\itemsep=3em
\item c
\end{itemize}

{\itemsep=2em
Itemsep outside has no effect:
\begin{itemize}
\item a
\item b
\item c
\end{itemize}}
