\section{chardef}
% Regular counter register.
% use the CS alone means assignment
\num=10\relax
% use with \the to get the numerical value
10 = \the\num

% Chardef register.
% use the CS alone gets the character
Lambda = \chara

% Use with \the to get the numerical value
3=\the\chara

% However, on RHS of assignment, without \the
\num=\chara
3=\the\num

\section{mathchardef}
